PROG -> DECLLIST PROCDECLLIST START ; STMTLIST END ;
DECLLIST -> ''
DECLLIST -> DECLSTMT ; DECLLIST
DECLSTMT -> DECLARE ( IDLIST ) DATATYPE
PROCDECLLIST -> ''
PROCDECLLIST -> PROCDECL PROCDECLLIST
PROCDECL -> id : PROCEDURE ( IDLIST ) ; STMTLIST END id ;
DATATYPE -> INTEGER
DATATYPE -> FLOAT
DATATYPE -> CHAR
DATATYPE -> ARRAYTYPE
ARRAYTYPE -> ARRAY [ E ] OF DATATYPE
STMTLIST -> ''
STMTLIST -> SUPERSTMT ; STMTLIST
SUPERSTMT -> LABELSTMT
SUPERSTMT -> IDLESSSTMT
LABELSTMT -> id POSTLABELSTMT
POSTLABELSTMT -> : STMT
POSTLABELSTMT -> ATTRSTMT
POSTLABELSTMT -> PROCSTMT
STMT -> LABELLESSSTMT
STMT -> IDLESSSTMT
LABELLESSSTMT -> id POSTLABELLESSSTMT
POSTLABELLESSSTMT -> ATTRSTMT 
POSTLABELLESSSTMT -> PROCSTMT
IDLESSSTMT -> STOPSTMT
IDLESSSTMT -> IOSTMT
IDLESSSTMT -> CONTROLSTMT
CONTROLSTMT -> IFSTMT
CONTROLSTMT -> GOTOSTMT
CONTROLSTMT -> LOOPSTMT
CONTROLSTMT -> EXITSTMT
IFSTMT -> IF E THEN STMTLIST ELSE1 ENDIF
ELSE1 -> ELSE STMTLIST
ELSE1 -> ''
GOTOSTMT -> GOTO id
LOOPSTMT -> LOOP ; STMTLIST ENDLOOP
EXITSTMT -> EXITWHEN E
STOPSTMT -> STOP
ATTRSTMT -> := E
PROCSTMT -> ( E )
E   -> T E'
E'  -> A E
E'  -> ''
A   -> +
A   -> -
T   -> !T
T   -> T2
T2  -> T3 T2'
T2' -> B T2
T2' -> ''
T3  -> T4 T3'
T3' -> C T3
T3' -> ''
C   -> >
C   -> <
C   -> =
C   -> !=
C   -> <=
C   -> >=
T4  -> F T4'
T4' -> D T4
T4' -> ''
D   -> *
D   -> /
D   -> %
F   -> - F
F   -> ( E )
F   -> id
